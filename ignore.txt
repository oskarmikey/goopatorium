package main

import (
	"fmt"
	"log"
	"time"

	"fyne.io/fyne/v2"
	"fyne.io/fyne/v2/app"
	"fyne.io/fyne/v2/container"
	"fyne.io/fyne/v2/widget"
)

func main() {
	myApp := app.New()
	myWindow := myApp.NewWindow(" Button testi")

	fpsLabel := widget.NewLabel("fps: 0")

	go func() {
		for {

			start := time.Now()
			time.Sleep(time.Millisecond * 100)
			fps := 1.0 / time.Since(start).Seconds()
			val := fmt.Sprintf("FPS: %.0f", fps)
			fpsLabel.SetText(val)
		}
	}()

	fmt.Println("this shit works!")

	input := widget.NewEntry()
	input.SetPlaceHolder("yolo")

	input_saveBtnSize := fyne.NewSize(200, 50)
	saveBtn := widget.NewButton("save", func() {
		log.Println("saved info is: ", input.Text)
	})
	sizedSaveBtn := container.NewGridWrap(input_saveBtnSize, saveBtn)

	boopBtn := widget.NewButton("boops here", func() {
		log.Println("Boop!")
	})

	mainLayout := container.NewVBox(
		fpsLabel,
		input,
		sizedSaveBtn,
		boopBtn,
	)

	myWindow.CenterOnScreen()

	myWindow.Canvas().Size()

	myWindow.SetContent(mainLayout)
	myWindow.ShowAndRun()

}
