package main

import (
	"fmt"
	"log"
	"time"

	"fyne.io/fyne/v2"
	"fyne.io/fyne/v2/app"
	"fyne.io/fyne/v2/container"
	"fyne.io/fyne/v2/widget"
)

func main() {
	myApp := app.New()
	myWindow := myApp.NewWindow("Button Test")

	fpsLabel := widget.NewLabel("fps: 0")

	// FPS counter goroutine
	go func() {
		for {
			start := time.Now()
			time.Sleep(time.Millisecond * 100)
			fps := 1.0 / time.Since(start).Seconds()
			val := fmt.Sprintf("FPS: %.0f", fps)
			fpsLabel.SetText(val)
		}
	}()

	input := widget.NewEntry()
	input.SetPlaceHolder("Enter text here")

	saveBtn := widget.NewButton("Save", func() {
		log.Println("Saved info:", input.Text)
	})

	boopBtn := widget.NewButton("Boop", func() {
		log.Println("Boop!")
	})

	mainLayout := container.NewVBox(
		fpsLabel,
		input,
		saveBtn,
		boopBtn,
	)

	myWindow.CenterOnScreen()
	myWindow.SetContent(mainLayout)
	myWindow.ShowAndRun()
}
